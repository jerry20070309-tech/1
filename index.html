// --- 僅在全域新增這一行記憶體 --- let peakPrices = {}; async function scan() { if(!isRunning) return; try { const [tRes, pRes] = await Promise.all([ fetch('https://fapi.binance.com/fapi/v1/ticker/24hr'), fetch('https://fapi.binance.com/fapi/v1/premiumIndex') ]); const tickers = await tRes.json(); const premiums = await pRes.json(); const premiumMap = Object.fromEntries(premiums.map(p => [p.symbol, p])); const now = new Date(); if (now.getMinutes() % 5 === 0 && now.getMinutes() !== lastSnapMin) { lastSnapMin = now.getMinutes(); tickers.forEach(t => { priceSnapshots[t.symbol] = parseFloat(t.lastPrice); }); document.getElementById('klineStatus').innerText = `${now.getHours()}:${now.getMinutes()} 5M對齊`; } tickers.forEach(t => { const s = t.symbol; if (!s.endsWith('USDT') || CONFIG.EXCLUDE.includes(s)) return; const price = parseFloat(t.lastPrice); const avgPrice = parseFloat(t.weightedAvgPrice); const fund = parseFloat(premiumMap[s]?.lastFundingRate || 0); const vol = parseFloat(t.quoteVolume); // 【核心修正 A】：記錄最高價，讓止盈損能「跟著價格走」 if (!peakPrices[s] || price > peakPrices[s]) peakPrices[s] = price; const mid24h = (parseFloat(t.highPrice) + parseFloat(t.lowPrice)) / 2; const m25 = (mid24h * 0.7) + (avgPrice * 0.3); const oldPrice = priceSnapshots[s] || price; const v5m = (price - oldPrice) / oldPrice; const isVolatile = v5m >= 0.05; if (!m7History[s]) m7History[s] = []; m7History[s].push(avgPrice); if (m7History[s].length > 10) m7History[s].shift(); const m7Slope = m7History[s].length > 1 ? avgPrice > m7History[s][0] : false; const m7Distance = (price - avgPrice) / avgPrice; const passStage1 = price > (avgPrice * 1.002); const passStage2 = m7Slope; const passStage3 = m7Distance < 0.08; if (!signalCount[s]) signalCount[s] = 0; if (passStage1 && passStage2 && passStage3) signalCount[s]++; else signalCount[s] = 0; const isAboveM7Filtered = signalCount[s] >= 3; const maSqz = (parseFloat(t.highPrice) - parseFloat(t.lowPrice)) / avgPrice; const drvScore = Math.min(100, (Math.abs(parseFloat(t.priceChangePercent)) / (vol / 1e8)) * 15); const isTight = Math.abs(price - avgPrice) / avgPrice < 0.003; const isAboveM99 = parseFloat(t.priceChangePercent) > 0; const isPocOk = Math.abs(parseFloat(t.priceChangePercent)) / 100 >= FILTER.POC; // 【核心修正 B】：將 TSL/TTP 與最高價 (peakPrices) 掛鉤，不再死板 let tsl, tslMode; if (drvScore > 90) { tsl = Math.max(avgPrice * 0.997, peakPrices[s] * 0.995); tslMode = "M7_TIGHT"; } else { tsl = Math.max(m25 * 0.992, peakPrices[s] * 0.985); tslMode = "M25_SAFE"; } let ttp = peakPrices[s] * (1 - FILTER.REV); let score = 0; if (isVolatile) score += 50; if (maSqz < 0.008) { score += (maSqz <= FILTER.SQZ) ? 25 : 0; score += (isPocOk) ? 25 : 0; score += (drvScore >= FILTER.DRV) ? 35 : 0; } else { score += (drvScore >= FILTER.DRV) ? 25 : 0; score += (isAboveM7Filtered && isAboveM99) ? 15 : 0; } stats[s] = { s, price, fund, maSqz, drvScore, isAboveM7: isAboveM7Filtered, isAboveM99, isTight, isPocOk, isVolatile, v5m, score, patterns:[], ttp, tsl, tslMode, peak: peakPrices[s] }; if ((score >= 90 || isVolatile) && (!lastPushTime[s] || Date.now() - lastPushTime[s] > CONFIG.COOLDOWN)) { lastPushTime[s] = Date.now(); addLog(s, v5m, score, isVolatile ? 2 : 1); } }); render(); } catch (e) { console.error(e); } setTimeout(scan, 3000); }
